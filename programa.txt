produtos=[]
# 1. CREATE (Adicionar)

def adicionar_produto(nome):
    produtos.append(nome)
    print(f" Produto '{nome}' adicionado com sucesso!")



# 2. READ (Listar)

def listar_produtos():
    if not produtos:
        print("A lista de produtos está vazia.")
        return

    print("\n--- LISTA DE PRODUTOS ---")
    for i, nome in enumerate(produtos, start=1):
        print(f"{i}. {nome}")
    print("-------------------------")

# 3. UPDATE (Atualizar)

def atualizar_produto(nome_antigo, nome_novo):
    if nome_antigo in produtos:
        # Descobre a posição (índice) da primeira ocorrência do produto
        indice = produtos.index(nome_antigo)
        produtos[indice] = nome_novo
        print(f" Produto '{nome_antigo}' atualizado para '{nome_novo}'!")
    else:
        print(f" Erro: Produto '{nome_antigo}' não encontrado.")


# 4. DELETE (Deletar)

def deletar_produto(nome):
    if nome in produtos:
        produtos.remove(nome)
        print(f" Produto '{nome}' removido com sucesso!")
    else:
        print(f" Erro: Produto '{nome}' não encontrado.")
adicionar_produto("smartphone")
adicionar_produto("fogao dako")
adicionar_produto("xiome")

produto = input("adicione um produto: ")
adicionar_produto(produto)
listar_produtos()

produto_antigo = input("Digite o nome do produto que quer atualizar: ")
produto_novo = input("Digite o novo nome: ")
atualizar_produto(produto_antigo, produto_novo)
listar_produtos()


produto = input("remova um produto: ")
deletar_produto(produto)
listar_produtos()
